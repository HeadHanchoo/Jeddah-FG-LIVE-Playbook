rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    match /customPlays/{playId} {
      allow read: if true;

      allow create: if
        request.resource.data.keys().hasOnly([
          'name',
          'description',
          'creatorName',
          'creatorNumber',
          'formation',
          'designerMode',
          'players',
          'likes',
          'likeCount',
          'createdAt'
        ])
        && request.resource.data.name is string
        && request.resource.data.name.size() > 0
        && request.resource.data.name.size() <= 80
        && request.resource.data.description is string
        && request.resource.data.description.size() <= 500
        && request.resource.data.creatorName is string
        && request.resource.data.creatorNumber is string
        && request.resource.data.players is list
        && request.resource.data.likeCount == 0;

      allow update: if
        request.resource.data.diff(resource.data).changedKeys().hasOnly(['likes', 'likeCount'])
        && request.resource.data.likeCount == resource.data.likeCount + 1;
    }
  }
}
